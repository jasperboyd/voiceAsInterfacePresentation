voicecommands.js

            var api = impress();
            api.init();

            function startVoice(){
                if (annyang) {
                
                    var commands = {
                        'next': function() {
                            api.next();
                        },
                        'previous': function() {
                            api.prev();
                        },
                    };

                    annyang.addCommands(commands);
                    annyang.start({ autoRestart: true, continuous: true });
                }
            }
